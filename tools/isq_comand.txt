SELECT
    E.EXP_DATA AS "Data",
    E.CLI_CODIGO AS "Cod Cliente",
    E.CLI_NOME AS "Cliente",
    E.PROD_CODIGO AS "Cod Produto",
    P.PROD_DESCRICAOPRODUTO AS "Descricao",
    E.EXP_QTDE AS "Qtde",
    E.EXP_VALORUNITARIO AS "Vr Unitario",
    (E.EXP_QTDE * E.EXP_VALORUNITARIO) AS "Vr Total",
    E.EXP_ENTRADASAIDA AS "Tipo",
    E.VEND_CODIGO AS "Cod Vendedor"
FROM
    EXTRATOPRODUTO E
JOIN 
    PRODUTO P ON E.PROD_CODIGO = P.PROD_CODIGO;
